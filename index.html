    <script>
        // Валидация формы
        document.getElementById('signupForm').addEventListener('submit', function(e) {
            let isValid = true;
            document.querySelectorAll('.error').forEach(el => el.textContent = '');
            document.querySelectorAll('input, select').forEach(el => el.classList.remove('invalid'));

            const name = document.getElementById('name');
            if (!/^[А-Яа-яЁё\s]+$/.test(name.value.trim())) {
                document.getElementById('nameError').textContent = 'Только русские буквы';
                name.classList.add('invalid');
                isValid = false;
            }

            const phone = document.getElementById('napassworme');
            if (!/^\+7\(\d{3}\)-\d{3}-\d{2}-\d{2}$/.test(phone.value.trim())) {
                document.getElementById('phoneError').textContent = 'Формат: +7(XXX)-XXX-XX-XX';
                phone.classList.add('invalid');
                isValid = false;
            }

            const email = document.getElementById('email');
            if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email.value.trim())) {
                document.getElementById('emailError').textContent = 'Неверный email';
                email.classList.add('invalid');
                isValid = false;
            }

            const corse = document.getElementById('corse');
            if (corse.value === '') {
                document.getElementById('corseError').textContent = 'Выберите курс';
                corse.classList.add('invalid');
                isValid = false;
            }

            if (!isValid) e.preventDefault();
        });
    </script>

</body>
</html>

🐘 5. PHP обработчик (test.php)

<?php
// 1. ПОДКЛЮЧЕНИЕ К БД
$sql = mysqli_connect("localhost", "root", "root", "user");
if (!$sql) { 
    die('Ошибка подключения: ' . mysqli_connect_error()); 
}

// 2. ПОЛУЧЕНИЕ ДАННЫХ ИЗ ФОРМЫ
// ВАЖНО: имена должны совпадать с name="" в HTML!
$nameUser = $_POST['name'];
$passwordUser = $_POST['napassworme'];  // это телефон
$emailUser = $_POST['email'];
$corseUser = $_POST['corse'];

// 3. ПОДГОТОВКА SQL-ЗАПРОСА (защита от SQL-инъекций)
$query = 'INSERT INTO user (name, password, email, corse) VALUES (?,?,?,?)';
$stmt = mysqli_prepare($sql, $query);

// 4. ПРИВЯЗКА ПАРАМЕТРОВ
// s = string, i = integer, d = double, b = blob
mysqli_stmt_bind_param($stmt, "ssss", $nameUser, $passwordUser, $emailUser, $corseUser);

// 5. ВЫПОЛНЕНИЕ ЗАПРОСА
$result = mysqli_stmt_execute($stmt);
if (!$result) { 
    print("Ошибка запроса"); 
}

// 6. ПЕРЕНАПРАВЛЕНИЕ НА ГЛАВНУЮ
header('Location: test.html');
exit();
?>
